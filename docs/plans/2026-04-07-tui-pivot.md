# TUI Pivot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Transform skill-unit from a skill-driven workflow into a dual-use CLI/TUI npm package with TypeScript, Ink, Citty, and a full CI/CD pipeline.

**Architecture:** Three-layer design: `src/core/` (business logic, no UI deps), `src/cli/` (Citty commands, TTY detection), `src/tui/` (Ink screens). Core engine ported from existing JS with event-emitter interfaces for live streaming. Stats system uses JSON-per-run with a rebuildable index.

**Tech Stack:** TypeScript (strict), Node.js 18+, Ink (TUI), Citty (CLI), Vitest (testing), marked + marked-terminal (markdown), tsx (dev runner)

**Design Spec:** `docs/specs/2026-04-07-tui-pivot-design.md`

```typescript
import fs from 'node:fs';
import type { SkillUnitConfig } from '../types/config.js';

export const CONFIG_DEFAULTS: SkillUnitConfig = {
  'test-dir': 'skill-tests',
  runner: {
    tool: 'claude',
    model: null,
    'max-turns': 10,
    'runner-concurrency': 5,
  },
  output: {
    format: 'interactive',
    'show-passing-details': false,
    'log-level': 'info',
  },
  execution: {
    timeout: '120s',
    'grader-concurrency': 5,
  },
  defaults: {
    setup: 'setup.sh',
    teardown: 'teardown.sh',
  },
};

export function parseYaml(text: string): Record<string, unknown> {
  const result: Record<string, unknown> = {};
  const lines = text.split('\n');
  let currentKey: string | null = null;
  let currentList: string[] | null = null;
  let currentObject: Record<string, unknown> | null = null;

  for (const rawLine of lines) {
    const line = rawLine.replace(/#.*$/, '').trimEnd();
    if (!line.trim()) continue;

    const indent = line.length - line.trimStart().length;

    // Block list item under a key
    if (indent > 0 && line.trimStart().startsWith('- ')) {
      if (currentKey && currentList) {
        currentList.push(parseScalar(line.trimStart().slice(2).trim()));
      }
      continue;
    }

    // Nested key-value under a parent key
    if (indent > 0 && currentKey && currentObject) {
      const colonIdx = line.indexOf(':');
      if (colonIdx !== -1) {
        const k = line.slice(0, colonIdx).trim();
        const v = line.slice(colonIdx + 1).trim();
        currentObject[k] = parseValue(v);
      }
      continue;
    }

    // Flush pending list or object
    if (currentKey) {
      if (currentList) {
        result[currentKey] = currentList;
        currentList = null;
      } else if (currentObject) {
        result[currentKey] = currentObject;
        currentObject = null;
      }
      currentKey = null;
    }

    // Top-level key: value
    const colonIdx = line.indexOf(':');
    if (colonIdx === -1) continue;

    const key = line.slice(0, colonIdx).trim();
    const valuePart = line.slice(colonIdx + 1).trim();

    if (!valuePart) {
      // No value: next lines are either a block list or nested object
      currentKey = key;
      currentList = null;
      currentObject = null;
      // Peek ahead to determine type
      const nextIdx = lines.indexOf(rawLine) + 1;
      if (nextIdx < lines.length) {
        const nextLine = lines[nextIdx].trimStart();
        if (nextLine.startsWith('- ')) {
          currentList = [];
        } else {
          currentObject = {};
        }
      }
      continue;
    }

    result[key] = parseValue(valuePart);
  }

  // Flush final pending
  if (currentKey) {
    if (currentList) result[currentKey] = currentList;
    else if (currentObject) result[currentKey] = currentObject;
  }

  return result;
}

function parseValue(v: string): unknown {
  if (v.startsWith('[') && v.endsWith(']')) {
    return v
      .slice(1, -1)
      .split(',')
      .map((s) => parseScalar(s.trim()));
  }
  return parseScalar(v);
}

function parseScalar(v: string): unknown {
  if (v === 'true') return true;
  if (v === 'false') return false;
  if (v === 'null') return null;
  if (v === '') return '';
  const num = Number(v);
  if (!isNaN(num) && v !== '') return num;
  // Strip surrounding quotes
  if (
    (v.startsWith('"') && v.endsWith('"')) ||
    (v.startsWith("'") && v.endsWith("'"))
  ) {
    return v.slice(1, -1);
  }
  return v;
}

function deepMerge(
  target: Record<string, unknown>,
  source: Record<string, unknown>
): Record<string, unknown> {
  const result = { ...target };
  for (const key of Object.keys(source)) {
    if (
      typeof source[key] === 'object' &&
      source[key] !== null &&
      !Array.isArray(source[key]) &&
      typeof result[key] === 'object' &&
      result[key] !== null &&
      !Array.isArray(result[key])
    ) {
      result[key] = deepMerge(
        result[key] as Record<string, unknown>,
        source[key] as Record<string, unknown>
      );
    } else {
      result[key] = source[key];
    }
  }
  return result;
}

export function loadConfig(configPath: string): SkillUnitConfig {
  if (!fs.existsSync(configPath)) {
    return structuredClone(CONFIG_DEFAULTS);
  }

  const raw = fs.readFileSync(configPath, 'utf-8');
  const parsed = parseYaml(raw);
  return deepMerge(
    CONFIG_DEFAULTS as unknown as Record<string, unknown>,
    parsed
  ) as unknown as SkillUnitConfig;
}
```

```typescript
import fs from 'node:fs';
import path from 'node:path';
import type { Spec, SpecFilter } from '../types/spec.js';

export function discoverSpecPaths(testDir: string): string[] {
  const results: string[] = [];
  walkDir(testDir, results);
  return results.sort();
}

function walkDir(dir: string, results: string[]): void {
  if (!fs.existsSync(dir)) return;

  const entries = fs.readdirSync(dir, { withFileTypes: true });
  for (const entry of entries) {
    const fullPath = path.join(dir, entry.name);
    if (entry.isDirectory()) {
      if (entry.name === 'fixtures') continue;
      walkDir(fullPath, results);
    } else if (entry.name.endsWith('.spec.md')) {
      results.push(fullPath);
    }
  }
}

export function filterSpecs(specs: Spec[], filter: SpecFilter): Spec[] {
  let result = specs;

  if (filter.name && filter.name.length > 0) {
    result = result.filter((s) => filter.name!.includes(s.frontmatter.name));
  }

  if (filter.tag && filter.tag.length > 0) {
    result = result.filter((s) =>
      s.frontmatter.tags.some((t) => filter.tag!.includes(t))
    );
  }

  if (filter.file && filter.file.length > 0) {
    result = result.filter((s) => {
      const normalized = path.normalize(s.path);
      return filter.file!.some((f) => normalized === path.normalize(f));
    });
  }

  if (filter.test && filter.test.length > 0) {
    result = result
      .map((s) => ({
        ...s,
        testCases: s.testCases.filter((tc) => filter.test!.includes(tc.id)),
      }))
      .filter((s) => s.testCases.length > 0);
  }

  return result;
}
```

```typescript
export function createEmptyIndex(): StatsIndex {
  return {

      totalTests: 0,
      passRate: 0,
      totalCost: 0,

export function loadIndex(baseDir: string): StatsIndex {
  const indexPath = path.join(baseDir, 'index.json');
  if (!fs.existsSync(indexPath)) {
    return createEmptyIndex();
  }
  return JSON.parse(fs.readFileSync(indexPath, 'utf-8'));

function saveIndex(baseDir: string, index: StatsIndex): void {
  index.lastUpdated = new Date().toISOString();
  const indexPath = path.join(baseDir, 'index.json');
  fs.mkdirSync(baseDir, { recursive: true });
  fs.writeFileSync(indexPath, JSON.stringify(index, null, 2));

export function recordRun(result: RunResult, baseDir: string): void {
  // Save run artifact
  const runDir = path.join(baseDir, 'runs', result.id);
  fs.mkdirSync(runDir, { recursive: true });
  fs.writeFileSync(
    path.join(runDir, 'run.json'),
    JSON.stringify(result, null, 2)
  );

  // Update index
  const index = loadIndex(baseDir);

  index.runs.push({
    id: result.id,
    timestamp: result.timestamp,
    testCount: result.testCount,
    passed: result.passed,
    failed: result.failed,
    duration: result.durationMs,
    cost: result.cost,
    tokens: result.tokens,
  });

  // Update per-test stats
  for (const test of result.tests) {
    const key = `${test.specName}/${test.id}`;
    const existing = index.tests[key];

    if (existing) {
      existing.runCount += 1;
      existing.passCount += test.passed ? 1 : 0;
      existing.avgDuration =
        (existing.avgDuration * (existing.runCount - 1) + test.durationMs) /
        existing.runCount;
      existing.lastRun = result.timestamp;
      existing.lastResult = test.passed ? 'pass' : 'fail';
    } else {
      index.tests[key] = {
        name: test.name,
        runCount: 1,
        passCount: test.passed ? 1 : 0,
        avgDuration: test.durationMs,
        avgCost: 0,
        avgTokens: 0,
        lastRun: result.timestamp,
        lastResult: test.passed ? 'pass' : 'fail',
      };
    }
  }

  // Update aggregate
  const agg = index.aggregate;
  agg.totalRuns = index.runs.length;
  agg.totalTests += result.testCount;
  agg.totalCost += result.cost;
  agg.totalTokens += result.tokens;

  const totalPassed = Object.values(index.tests).reduce(
    (sum, t) => sum + t.passCount,
    0
  );
  const totalTestRuns = Object.values(index.tests).reduce(
    (sum, t) => sum + t.runCount,
    0
  );
  agg.passRate = totalTestRuns > 0 ? totalPassed / totalTestRuns : 0;

  saveIndex(baseDir, index);
}

export function cleanupRuns(baseDir: string, keepCount: number): void {
  const index = loadIndex(baseDir);
  const runsDir = path.join(baseDir, 'runs');

  if (index.runs.length <= keepCount) return;

  const toRemove = index.runs.slice(0, index.runs.length - keepCount);
  index.runs = index.runs.slice(index.runs.length - keepCount);

  for (const run of toRemove) {
    const runDir = path.join(runsDir, run.id);
    if (fs.existsSync(runDir)) {
      fs.rmSync(runDir, { recursive: true, force: true });
    }
  }

  index.tests = {};
  for (const runEntry of index.runs) {
    const runDir = path.join(runsDir, runEntry.id);
    const runDataPath = path.join(runDir, 'run.json');

        };
      }
    }
  }

  agg.totalTests = index.runs.reduce((sum, r) => sum + r.testCount, 0);

export function rebuildIndex(baseDir: string): void {
```

```tsx
import React, { useState, useMemo } from 'react';
import { Box, Text, useInput } from 'ink';
import type { Spec, TestCase } from '../../types/spec.js';

interface FlatTest {
  specName: string;
  specPath: string;
  tags: string[];
  testCase: TestCase;
  selected: boolean;
}

interface DashboardProps {
  specs: Spec[];
  onRunTests: (tests: FlatTest[]) => void;
}

export function Dashboard({ specs, onRunTests }: DashboardProps) {
  const [query, setQuery] = useState('');
  const [cursor, setCursor] = useState(0);
  const [selected, setSelected] = useState<Set<string>>(new Set());

  const allTests = useMemo(() => {
    const tests: FlatTest[] = [];
    for (const spec of specs) {
      for (const tc of spec.testCases) {
        tests.push({
          specName: spec.frontmatter.name,
          specPath: spec.path,
          tags: spec.frontmatter.tags,
          testCase: tc,
          selected: selected.has(`${spec.frontmatter.name}/${tc.id}`),
        });
      }
    }
    return tests;
  }, [specs, selected]);

  const filtered = useMemo(() => {
    if (!query.trim()) return allTests;

    const parts = query.trim().split(/\s+/);
    let result = allTests;

    for (const part of parts) {
      if (part.startsWith('tag:')) {
        const tag = part.slice(4);
        result = result.filter((t) => t.tags.some((tg) => tg.includes(tag)));
      } else {
        const lower = part.toLowerCase();
        result = result.filter(
          (t) =>
            t.testCase.name.toLowerCase().includes(lower) ||
            t.testCase.id.toLowerCase().includes(lower)
        );
      }
    }
    return result;
  }, [allTests, query]);

  useInput((input, key) => {
    if (key.upArrow) setCursor((c) => Math.max(0, c - 1));
    if (key.downArrow) setCursor((c) => Math.min(filtered.length - 1, c + 1));
    if (input === ' ' && filtered[cursor]) {
      const key = `${filtered[cursor].specName}/${filtered[cursor].testCase.id}`;
      setSelected((prev) => {
        const next = new Set(prev);
        if (next.has(key)) next.delete(key);
        else next.add(key);
        return next;
      });
    }
    if (input === 'a') {
      if (selected.size === filtered.length) {
        setSelected(new Set());
      } else {
        setSelected(
          new Set(filtered.map((t) => `${t.specName}/${t.testCase.id}`))
        );
      }
    }
    if (key.return) {
      const testsToRun = filtered.filter((t) =>
        selected.has(`${t.specName}/${t.testCase.id}`)
      );
      if (testsToRun.length > 0) onRunTests(testsToRun);
    }
  });

  const selectedCount = selected.size;

  return (
    <Box flexDirection="column">
      <Box marginBottom={1}>
        <Text color="blue">{'> '}</Text>
        <Text>{query || 'Search tests...'}</Text>
      </Box>

      <Box marginBottom={1}>
        <Text color="gray">
          {filtered.length} test(s){' '}
          {selectedCount > 0 ? `| ${selectedCount} selected` : ''}
        </Text>
      </Box>

      {filtered.map((t, i) => {
        const isSelected = selected.has(`${t.specName}/${t.testCase.id}`);
        const isCursor = i === cursor;
        return (
          <Box key={`${t.specName}/${t.testCase.id}`}>
            <Text color={isCursor ? 'blue' : undefined}>
              {isCursor ? '>' : ' '} {isSelected ? '[x]' : '[ ]'}{' '}
              {t.testCase.id}: {t.testCase.name}
            </Text>
            <Text color="gray"> ({t.specName})</Text>
            {t.tags.length > 0 && (
              <Text color="yellow"> [{t.tags.join(', ')}]</Text>
            )}
          </Box>
        );
      })}

      {filtered.length === 0 && (
        <Text color="gray">No tests match your search.</Text>
      )}
    </Box>
  );
}
```

```tsx
import React from 'react';
import { Box, Text } from 'ink';
import type { TestStatus } from '../../types/run.js';

interface ProgressTest {
  id: string;
  name: string;
  status: TestStatus;
  durationMs: number;
}

interface ProgressTreeProps {
  tests: ProgressTest[];
  elapsed: number;
}

const STATUS_ICONS: Record<TestStatus, { icon: string; color: string }> = {
  pending: { icon: '\u25CB', color: 'gray' }, // ○
  running: { icon: '\u23F3', color: 'blue' }, // ⏳
  grading: { icon: '\u2699', color: 'yellow' }, // ⚙
  passed: { icon: '\u2714', color: 'green' }, // ✔
  failed: { icon: '\u2718', color: 'red' }, // ✘
  timedout: { icon: '\u23F0', color: 'red' }, // ⏰
  error: { icon: '\u2718', color: 'red' }, // ✘
};

export function ProgressTree({ tests, elapsed }: ProgressTreeProps) {
  const completed = tests.filter((t) =>
    ['passed', 'failed', 'timedout', 'error'].includes(t.status)
  ).length;
  const elapsedStr = formatElapsed(elapsed);

  return (
    <Box flexDirection="column" width={30}>
      <Text bold color="blue">
        PROGRESS ({completed}/{tests.length})
      </Text>
      <Box flexDirection="column" marginTop={1}>
        {tests.map((t) => {
          const { icon, color } = STATUS_ICONS[t.status];
          return (
            <Box key={t.id}>
              <Text color={color}>{icon}</Text>
              <Text bold={t.status === 'running'}> {t.name} </Text>
              <Text color="gray">
                {t.status === 'running'
                  ? 'running'
                  : t.status === 'grading'
                    ? 'grading'
                    : t.status === 'pending'
                      ? ''
                      : `${(t.durationMs / 1000).toFixed(1)}s`}
              </Text>
            </Box>
          );
        })}
      </Box>
      <Box marginTop={1} flexDirection="column">
        <Text color="gray">Elapsed: {elapsedStr}</Text>
      </Box>
    </Box>
  );
}

function formatElapsed(ms: number): string {
  const secs = Math.floor(ms / 1000);
  const mins = Math.floor(secs / 60);
  const s = secs % 60;
  return `${String(mins).padStart(2, '0')}:${String(s).padStart(2, '0')}`;
}
```

```typescript
import { useState, useEffect, useCallback, useRef } from 'react';
import type { TestStatus, RunProgress } from '../../types/run.js';

export interface TestRunState {
  status: 'idle' | 'running' | 'complete';
  tests: Array<{
    id: string;
    name: string;
    specName: string;
    status: TestStatus;
    durationMs: number;
    transcript: string[];
    activity: string;
  }>;
  activeTestId: string | null;
  elapsed: number;
}

export function useTestRun() {
  const [state, setState] = useState<TestRunState>({
    status: 'idle',
    tests: [],
    activeTestId: null,
    elapsed: 0,
  });

  const timerRef = useRef<ReturnType<typeof setInterval>>();
  const startTimeRef = useRef<number>(0);

  const startRun = useCallback(
    (tests: Array<{ id: string; name: string; specName: string }>) => {
      startTimeRef.current = Date.now();
      setState({
        status: 'running',
        tests: tests.map((t) => ({
          ...t,
          status: 'pending' as TestStatus,
          durationMs: 0,
          transcript: [],
          activity: '',
        })),
        activeTestId: tests[0]?.id || null,
        elapsed: 0,
      });

      timerRef.current = setInterval(() => {
        setState((prev) => ({
          ...prev,
          elapsed: Date.now() - startTimeRef.current,
        }));
      }, 1000);
    },
    []
  );

  const selectTest = useCallback((testId: string) => {
    setState((prev) => ({ ...prev, activeTestId: testId }));
  }, []);

  const updateTest = useCallback(
    (testId: string, update: Partial<TestRunState['tests'][0]>) => {
      setState((prev) => ({
        ...prev,
        tests: prev.tests.map((t) =>
          t.id === testId ? { ...t, ...update } : t
        ),
      }));
    },
    []
  );

  const completeRun = useCallback(() => {
    if (timerRef.current) clearInterval(timerRef.current);
    setState((prev) => ({ ...prev, status: 'complete' }));
  }, []);

  useEffect(() => {
    return () => {
      if (timerRef.current) clearInterval(timerRef.current);
    };
  }, []);

  return { state, startRun, selectTest, updateTest, completeRun };
}
```

```tsx
import { Box, useInput } from 'ink';

export function Runner({ runState, onSelectTest }: RunnerProps) {

      <ProgressTree tests={tests} elapsed={elapsed} />
      <Box flexDirection="column" flexGrow={1} marginLeft={1}>

              status: s.status,
```

```typescript
import fs from 'node:fs';
import path from 'node:path';

export interface SelectionState {
  selectedTests: Set<string>;
  viewMode: 'primary' | 'split';
}

export function loadSelection(baseDir: string): SelectionState {
  const filePath = path.join(baseDir, 'selection.json');
  if (!fs.existsSync(filePath)) {
    return { selectedTests: new Set(), viewMode: 'primary' };
  }
  const data = JSON.parse(fs.readFileSync(filePath, 'utf-8'));
  return {
    selectedTests: new Set(data.selectedTests || []),
    viewMode: data.viewMode || 'primary',
  };
}

export function saveSelection(state: SelectionState, baseDir: string): void {
  fs.mkdirSync(baseDir, { recursive: true });
  const filePath = path.join(baseDir, 'selection.json');
  fs.writeFileSync(
    filePath,
    JSON.stringify(
      {
        selectedTests: [...state.selectedTests],
        viewMode: state.viewMode,
      },
      null,
      2
    )
  );
}
```
